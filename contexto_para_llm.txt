def load_config(): # carga la configuración desde variables de entorno
    """Carga y valida la configuración desde variables de entorno."""
    load_dotenv() # carga las variables de entorno

    config = {
        "backend": os.getenv("BACKEND", "ollama").lower(), # os.getenv() es una función que retorna el valor de una variable de entorno.Y
        # si no existe la variable de entorno, retorna el valor por defecto que le pasamos (Que en este caso es "ollama").
        "temperature": float(os.getenv("TEMPERATURE", "0.0")),
        "ollama_url": os.getenv("OLLAMA_BASE_URL", "http://localhost:11434/v1/chat/completions"),
        "ollama_model": os.getenv("OLLAMA_MODEL", "tinyllama"),
        "groq_url": os.getenv("GROQ_BASE_URL", "https://api.groq.com/openai/v1/chat/completions"),
        "groq_model": os.getenv("GROQ_MODEL", "llama-3.3-70b-versatile"),
        "groq_api_key": os.getenv("GROQ_API_KEY"),
    }

    if config["backend"] not in ["ollama", "groq"]: # 
        print(f"Error: BACKEND inválido '{config['backend']}'.")
        sys.exit(1)

    if config["backend"] == "groq" and not config["groq_api_key"]: # si el backend es groq y no hay api key
        print("Error: BACKEND=groq requiere GROQ_API_KEY.") 
        sys.exit(1) # sale del programa

    return config # retorna la configuración. Es decir, nos está dando las variables de entorno

def chat_backend(messages, config, tools=None):# ChatBackend es una función que recibe 3 parámetros: messages, config y tools.
    # tools es una lista de herramientas que se le pasan al chat y es None porque es opcional que utilice la herramienta.
    # messages es una lista de diccionarios que representan el historial de la conversación.
    # config es un diccionario que contiene la configuración del chat.
    """Función unificada para Ollama y Groq."""
    if config["backend"] == "groq":
        url = config["groq_url"]
        headers = {"Authorization": f"Bearer {config['groq_api_key']}", "Content-Type": "application/json"} 
        #Los headeres son las cabeceras que se le pasan al chat, en este caso le pasamos la api key y el tipo de contenido que estamos enviando (Si es una imagen o un json).
        model = config["groq_model"]
    else: # si el backend es ollama
        url = config["ollama_url"]
        headers = {"Content-Type": "application/json"} # Ollama no necesita autenticación de api key. 
        model = config["ollama_model"] # Le pasamos el modelo que queremos usar.
# payload es un diccionario que contiene la información que se le envía al chat.
    payload = {
        "model": model, # Le pasamos el modelo que queremos usar.
        "messages": messages,# Le pasamos el historial de la conversación.
        "temperature": config["temperature"], # La temperatura es un parámetro que controla la creatividad del modelo. Un valor de 0.0 hace que el modelo sea más determinista, mientras que un valor de 1.0 hace que sea más creativo.
        "stream": False, # Stream es un parámetro que controla si el modelo debe enviar la respuesta en tiempo real o en un solo bloque. Un valor de False hace que el modelo envíe la respuesta en un solo bloque.
    }
    
    if tools: # Si tools es True, entonces le pasamos las herramientas al payload.
        payload["tools"] = tools # Le pasamos las herramientas al payload.
        payload["tool_choice"] = "auto" # Tool_choice es un parámetro que controla si el modelo debe usar las herramientas. Un valor de "auto" hace que el modelo use las herramientas siempre que quiera
#Aquí es donde el código intenta comunicarse con el servidor (Ollama o Groq).
    try:
        #url: A dónde va el mensaje.
        #json=payload: El contenido (la carta).
        #headers=headers: Las etiquetas del sobre (el sello y el tipo de contenido).
        #timeout=60: Es el cronómetro. Si el servidor no responde en 60 segundos, el programa "se rinde" para no quedarse colgado para siempre.
        response = requests.post(url, json=payload, headers=headers, timeout=60) # Aquí es donde el código intenta comunicarse con el servidor (Ollama o Groq).
        response.raise_for_status() # Si la respuesta no es exitosa, se lanza una excepción.

        # --------------------------------------------

    #Printea el Json mandado y recibido

        full_json = response.json() # full_json es una variable que almacena la respuesta del servidor en formato json.

        print("\n--- RESPUESTA JSON DEL SERVIDOR ---")

        print(json.dumps(full_json, indent=2, ensure_ascii=False)) # json.dumps es una función que convierte un objeto json a una cadena de texto.

        print("------------------------------------\n")

    # --------------------------------------------
        return response.json()["choices"][0]["message"] 
        """El servidor de Ollama o Groq devuelve un objeto json. lo que hace esta línea es
        extraer el mendaje de esa respuesta"""
    except requests.exceptions.HTTPError as e:
        print(f"Error de conexión: {e}") # Si hay un error de conexión, se imprime el error.
        print(f"Detalle del error: {e.response.text}")  # Muestra el cuerpo del error para debuggear
        return None
    except requests.exceptions.RequestException as e:
        print(f"Error de conexión: {e}")
        return None

"""En el archivo mis_tools.py, tú definiste un contrato (un esquema JSON) para la función consultar_pdf.
Al poner "busqueda" dentro de properties, le estás diciendo a la IA: "Si quieres usar mi PDF, tienes que enviarme el texto dentro de una etiqueta llamada exactamente 'busqueda'".
Es como el campo "Asunto" en un correo electrónico; sin ese nombre, el receptor no sabe dónde mirar para entender de qué trata el mensaje.
2. Es el "Paquete" que envía la IA. Cuando el usuario hace una pregunta técnica, la IA (el LLM) no responde con texto normal, sino que crea un mensaje con una instrucción técnica.
La IA escribe algo como: {"busqueda": "normas de seguridad"}.
Aquí, "busqueda" es la clave y "normas de seguridad" es el valor (lo que el usuario quiere saber).
3. Es la "Variable" que usa tu código Python
En app.py, tu código recibe ese paquete de la IA. Como viene en formato de texto (JSON), necesitas extraer el contenido de la caja para poder usarlo:
args = json.loads(...): Esta línea abre el paquete.
query = args.get("busqueda"): Esta línea busca la etiqueta "busqueda", saca el valor que hay dentro (por ejemplo, "normas de seguridad") y lo guarda en la variable query.
4. Es el "Motor" de la búsqueda final
Finalmente, esa variable query (que sacaste de la clave "busqueda") es la que se envía a la función buscar_contexto en Rag.py.
Sin esa clave, tu función buscar_contexto recibiría un vacío y no sabría qué fragmentos del PDF buscar en la base de datos de ChromaDB."""

import pandas as pd
import os
import kagglehub
from sklearn.ensemble import RandomForestRegressor

print("🧠 Entrenando modelo de Machine Learning (House Pricing) para el bot...")
# Cargar datos
path = kagglehub.dataset_download("rukenmissonnier/final-house")
ruta_csv = os.path.join(path, 'house.csv')
df = pd.read_csv(ruta_csv)

# Separar y entrenar (Entrenamos con todos los datos para máxima precisión)
X = df.drop(columns=['price'])
y = df['price']

import os
import uuid
import chromadb
from chromadb.utils import embedding_functions
from pypdf import PdfReader

# ======================================================================================
# 🧠 CEREBRO DEL SISTEMA RAG "Retrieval-Augmented Generation" 
# ======================================================================================
"""
Este archivo es el responsable de la "Memoria a Largo Plazo" de tu IA.
Su misión es transformar un documento PDF estático en una base de datos "inteligente" (Semántica).

Funciona en 3 pasos:
1. LEER: Extrae el texto plano del PDF.
2. TROCEAR (Chunking): Divide el texto en fragmentos pequeños para que la IA pueda procesarlos.
3. VECTORIZAR Y GUARDAR: Convierte esos fragmentos en listas de números (embeddings) y los guarda
   en ChromaDB, permitiendo buscar información por significado y no solo por palabras clave.
"""

# --- 1. CLASES Y FUNCIONES AUXILIARES ---

class RecursiveChunker: 
    """
    ✂️ CLASE RECORTADORA DE TEXTO
    La IA tiene un límite de atención (ventana de contexto). No puede "leer" un libro entero de golpe.
    Esta clase se encarga de dividir el texto en trozos (chunks) manejables y lógicos.
    """
    def __init__(self, chunk_size=1000, chunk_overlap=100, separators=None):
        """
        Configuración de la "tijera":
        - chunk_size=1000: Tamaño objetivo de cada trozo (caracteres).
        - chunk_overlap=100: Solapamiento. Las últimas 100 letras de un trozo se repiten al inicio del siguiente.
          ¿Por qué? Para evitar cortar una frase o idea importante justo por la mitad y perder el contexto.
        - separators: Lista de preferencias de corte. Primero intenta cortar por párrafos (\n\n),
          si el trozo sigue siendo grande, prueba por líneas (\n), luego por espacios...
        """
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.separators = separators or ["\n\n", "\n", " ", ""]

    def split_text(self, text):
        # Función pública que inicia el proceso de división
        return self._split_text_recursive(text, self.separators)

    def _split_text_recursive(self, text, separators):
        # Lógica interna (Recursiva):
        # Si un fragmento es demasiado grande, se llama a sí misma con un separador más fino
        # hasta lograr el tamaño adecuado.
        final_chunks = []
        if not separators: return [text]
        separator = separators[0]
        next_separators = separators[1:]
        
        # Intentamos dividir con el separador actual (ej: párrafos)
        splits = list(text) if separator == "" else text.split(separator)
        
        current_doc = []
        current_length = 0
        
        for split in splits:
            len_sep = len(separator) if current_doc else 0
            split_len = len(split) + len_sep
            
            # Si añadir este trozo supera el tamaño máximo permitido...
            if current_length + split_len > self.chunk_size:
                if current_doc:
                    # ..cerramos el chunk actual y lo guardamos
                    doc_text = separator.join(current_doc)
                    if doc_text.strip(): final_chunks.append(doc_text)
                    
                    # Gestionamos el solapamiento (overlap) para el siguiente chunk
                    while current_length > self.chunk_overlap and current_doc:
                        removed = current_doc.pop(0)
                        current_length -= (len(removed) + len(separator))
                    if current_length < 0: current_length = 0
                
                # Si el trozo individual es GIGANTE (mayor que el límite), lo dividimos recursivamente
                if len(split) > self.chunk_size:
                    sub_chunks = self._split_text_recursive(split, next_separators)
                    final_chunks.extend(sub_chunks)
                else:
                    current_doc.append(split)
                    current_length += len(split) + (len(separator) if len(current_doc) > 1 else 0)
            else:
                # Si cabe, lo añadimos al chunk actual que estamos construyendo
                current_doc.append(split)
                current_length += split_len
        
        # Añadir lo que haya quedado pendiente
        if current_doc:
            doc_text = separator.join(current_doc)
            if doc_text.strip(): final_chunks.append(doc_text)
        return final_chunks

def cargar_pdf(ruta_archivo):
    """
    📖 Lector de PDF simple.
    Extrae todo el texto página a página y lo concatena en un solo string gigante.
    """
    if not os.path.exists(ruta_archivo):
        print(f"❌ Error: El archivo {ruta_archivo} no existe en la carpeta.")
        return None
    reader = PdfReader(ruta_archivo)
    texto = ""
    for page in reader.pages:
        extracted = page.extract_text()
        if extracted: texto += extracted + "\n"
    return texto

# --- 2. FUNCIÓN DE INICIALIZACIÓN (IMPORTANTE: PUNTO DE ENTRADA) ---
def inicializar_rag(lista_pdfs):
    """
    ⚙️ ARRANQUE DEL SISTEMA
    Comprueba si ya tenemos la base de datos vectorial creada en el disco.
    - SI EXISTE: La carga y listo.
    - NO EXISTE: Lee el PDF, lo procesa y crea la base de datos desde cero.
    """
    print("⚙️ Iniciando ChromaDB...")
    
    # Manejo de un solo PDF como string (por seguridad)
    if isinstance(lista_pdfs, str):
        lista_pdfs = [lista_pdfs]

    # A. Configurar el "Traductor a Números" (Embedding Function)
    emb_fn = embedding_functions.SentenceTransformerEmbeddingFunction(model_name="all-MiniLM-L6-v2")
    
    # B. Conectar a la Base de Datos (Persistente en disco)
    client = chromadb.PersistentClient(path="./chroma_db")
    
    # C. Obtener o Crear la Colección 
    collection = client.get_or_create_collection(name="mi_base_vectorial", embedding_function=emb_fn, metadata={"hnsw:space": "cosine"})

    # D. Indexado de documentos
    for pdf in lista_pdfs:
        # Verificamos si ESTE archivo específico ya está indexado usando metadatos
        # Buscamos si existe al menos 1 fragmento con source = pdf
        existing = collection.get(where={"source": pdf}, limit=1)
        
        if len(existing['ids']) == 0: 
            print(f"📖 Indexando nuevo documento: {pdf}...")
            texto = cargar_pdf(pdf)

            if texto:
                # 2. Trocear texto
                chunker = RecursiveChunker(chunk_size=1000, chunk_overlap=200)
                chunks = chunker.split_text(texto)
                
                # 3. Preparar metadatos e IDs
                ids = [str(uuid.uuid4()) for _ in range(len(chunks))]
                # Guardamos el nombre del archivo en los metadatos 'source'
                metadatos = [{"source": pdf} for _ in range(len(chunks))]
                
                # 4. Guardar en ChromaDB
                collection.add(documents=chunks, metadatas=metadatos, ids=ids)
                print(f"✅ Indexación completada: {len(chunks)} fragmentos de {pdf}.")
            else:
                print(f"⚠️ ERROR: No se pudo leer {pdf}. Asegúrate de que el archivo existe.")
        else:
            print(f"✅ {pdf} ya está indexado en la base de datos.")
    
    return collection

# --- 3. FUNCIÓN DE BÚSQUEDA ---
def buscar_contexto(collection, query):
    """
    🕵️‍♂️ EL BUSCADOR
    Recibe la pregunta del usuario y encuentra los párrafos del PDF más relacionados.
    Retorna un texto único con la información encontrada.
    """
    # Pedimos también los 'metadatas' para saber el nombre del archivo
    resultados = collection.query(query_texts=[query], n_results=3)
    
    contexto_formateado = ""
    
    if resultados['documents'] and resultados['documents'][0]:
        # Iteramos sobre los documentos y sus metadatos simultáneamente
        for doc, meta in zip(resultados['documents'][0], resultados['metadatas'][0]):
            fuente = meta.get('source', 'Desconocido')
            contexto_formateado += f"--- Fuente: {fuente} ---\n{doc}\n\n"
        return contexto_formateado
        
    return "No se encontró información relevante en los documentos."
